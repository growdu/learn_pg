// GetProject returns a single project by ID, or nil if not found.
func (s *workspaceStore) GetProject(id string) (*workspaceProject, error) {
	s.mu.Lock()
	defer s.mu.Unlock()
	env, err := s.readEnvelopeLocked()
	if err != nil {
		return nil, err
	}
	for i := range env.Projects {
		if env.Projects[i].ID == id {
			return &env.Projects[i], nil
		}
	}
	return nil, nil
}

// UpdateProjectLocked updates a project by ID (caller holds lock).
func (s *workspaceStore) UpdateProjectLocked(id string, patch func(*workspaceProject) error) error {
	env, err := s.readEnvelopeLocked()
	if err != nil {
		return err
	}
	for i := range env.Projects {
		if env.Projects[i].ID == id {
			if err := patch(&env.Projects[i]); err != nil {
				return err
			}
			return s.writeEnvelopeLocked(env)
		}
	}
	return fmt.Errorf("project not found: %s", id)
}

// DeleteProjectLocked deletes a project by ID (caller holds lock).
func (s *workspaceStore) DeleteProjectLocked(id string) error {
	env, err := s.readEnvelopeLocked()
	if err != nil {
		return err
	}
	next := make([]workspaceProject, 0, len(env.Projects))
	for _, p := range env.Projects {
		if p.ID != id {
			next = append(next, p)
		}
	}
	env.Projects = next
	return s.writeEnvelopeLocked(env)
}

func (h *Handler) ServeProjectCreate(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		h.writeError(w, r, http.StatusMethodNotAllowed, "POST required")
		return
	}
	body, err := io.ReadAll(http.MaxBytesReader(w, r.Body, 64*1024))
	if err != nil {
		h.writeError(w, r, http.StatusBadRequest, "failed to read body")
		return
	}
	defer r.Body.Close()

	var req ProjectCreateRequest
	if err := json.Unmarshal(body, &req); err != nil {
		h.writeError(w, r, http.StatusBadRequest, "invalid JSON")
		return
	}
	if strings.TrimSpace(req.ID) == "" {
		h.writeError(w, r, http.StatusBadRequest, "id is required")
		return
	}
	if strings.TrimSpace(req.Name) == "" {
		h.writeError(w, r, http.StatusBadRequest, "name is required")
		return
	}

	project := workspaceProject{
		ID:         req.ID,
		Name:       req.Name,
		Clusters:   []workspaceCluster{},
		Components: []workspaceComponent{},
	}
	if err := h.workspace.upsert(project); err != nil {
		h.writeError(w, r, http.StatusInternalServerError, err.Error())
		return
	}
	writeJSON(w, r, http.StatusCreated, map[string]any{"success": true, "id": req.ID})
}

// nodeIDFromPath extracts node ID from /api/nodes/{id}.
func nodeIDFromPath(path string) string {
	parts := strings.TrimPrefix(path, "/api/nodes/")
	i := strings.Index(parts, "/")
	if i >= 0 {
		return parts[:i]
	}
	return strings.TrimSpace(parts)
}
