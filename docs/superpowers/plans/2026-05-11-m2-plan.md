```go
package connection

import (
	"fmt"
	"sync"
	"sync/atomic"

	"pg-visualizer-backend/internal/config"
	"pg-visualizer-backend/internal/pg"
)

// Manager manages database connections keyed by nodeId.
type Manager struct {
	mu      sync.RWMutex
	conns   map[string]*pg.Client // nodeId -> active connection
	cfgStore map[string]*pg.Config // nodeId -> config for reconnect
	active  atomic.String          // current active nodeId
	config  *config.Config
}

// NewManager creates a new connection manager.
func NewManager(cfg *config.Config) *Manager {
	return &Manager{
		conns:    make(map[string]*pg.Client),
		cfgStore: make(map[string]*pg.Config),
		config:   cfg,
	}
}

// Config holds connection parameters for a node.
type Config struct {
	Host     string
	Port     int
	User     string
	Password string
	Database string
}

// Register adds or updates connection config for a node (does not connect).
func (m *Manager) Register(nodeId string, cfg Config) {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.cfgStore[nodeId] = &cfg
}

// Unregister removes a node's config and closes its connection.
func (m *Manager) Unregister(nodeId string) error {
	m.mu.Lock()
	defer m.mu.Unlock()
	if client, ok := m.conns[nodeId]; ok {
		client.Close()
		delete(m.conns, nodeId)
	}
	delete(m.cfgStore, nodeId)
	if m.active.Load() == nodeId {
		m.active.Store("")
	}
	return nil
}

// Get returns the connection for a node, or error if not connected.
func (m *Manager) Get(nodeId string) (*pg.Client, error) {
	m.mu.RLock()
	defer m.mu.RUnlock()
	client, ok := m.conns[nodeId]
	if !ok {
		return nil, fmt.Errorf("node %s: not connected", nodeId)
	}
	return client, nil
}

// Activate connects to a node and sets it as active.
func (m *Manager) Activate(nodeId string) error {
	m.mu.Lock()
	defer m.mu.Unlock()

	cfg, ok := m.cfgStore[nodeId]
	if !ok {
		return fmt.Errorf("node %s: config not found, call Register first", nodeId)
	}

	// Close existing connection for this node if any
	if client, ok := m.conns[nodeId]; ok {
		client.Close()
	}

	client := &pg.Client{}
	if err := client.Connect(cfg.Host, cfg.Port, cfg.User, cfg.Password, cfg.Database); err != nil {
		return fmt.Errorf("node %s: %w", nodeId, err)
	}
	m.conns[nodeId] = client
	m.active.Store(nodeId)
	return nil
}

// Deactivate closes connection for a node but keeps its config.
func (m *Manager) Deactivate(nodeId string) error {
	m.mu.Lock()
	defer m.mu.Unlock()
	if client, ok := m.conns[nodeId]; ok {
		client.Close()
		delete(m.conns, nodeId)
	}
	if m.active.Load() == nodeId {
		m.active.Store("")
	}
	return nil
}

// GetActive returns the active nodeId and its connection.
func (m *Manager) GetActive() (string, *pg.Client) {
	nodeId := m.active.Load()
	if nodeId == "" {
		return "", nil
	}
	m.mu.RLock()
	defer m.mu.RUnlock()
	return nodeId, m.conns[nodeId]
}

// Health checks if a node's connection is alive.
func (m *Manager) Health(nodeId string) (bool, error) {
	m.mu.RLock()
	client, ok := m.conns[nodeId]
	m.mu.RUnlock()
	if !ok {
		return false, fmt.Errorf("not connected")
	}
	if err := client.Ping(); err != nil {
		return false, err
	}
	return true, nil
}

// GetConfig returns the stored config for a node.
func (m *Manager) GetConfig(nodeId string) (Config, bool) {
	m.mu.RLock()
	defer m.mu.RUnlock()
	cfg, ok := m.cfgStore[nodeId]
	if !ok {
		return Config{}, false
	}
	return *cfg, true
}
```

// UpdateNodeStatus updates connection status and last error for a node.
func (s *workspaceStore) UpdateNodeStatus(nodeId string, status string, lastError string) error {
	s.mu.Lock()
	defer s.mu.Unlock()
	env, err := s.readEnvelopeNoLock()
	if err != nil {
		return err
	}
	for pi := range env.Projects {
		for ci := range env.Projects[pi].Clusters {
			for ni := range env.Projects[pi].Clusters[ci].Nodes {
				if env.Projects[pi].Clusters[ci].Nodes[ni].ID == nodeId {
					env.Projects[pi].Clusters[ci].Nodes[ni].ConnectionStatus = status
					env.Projects[pi].Clusters[ci].Nodes[ni].LastError = lastError
					return s.writeEnvelopeLocked(env)
				}
			}
		}
	}
	return fmt.Errorf("node not found: %s", nodeId)
}

// ServeNodeActivate handles POST /api/nodes/{id}/activate.
// Activates a node: reads config from workspace, connects, sets as active.
func (h *Handler) ServeNodeActivate(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		h.writeError(w, r, http.StatusMethodNotAllowed, "POST required")
		return
	}
	nodeID := extractNodeID(r.URL.Path)
	if nodeID == "" {
		h.writeError(w, r, http.StatusBadRequest, "node id required")
		return
	}

	// Find the node in workspace to get connection config
	projects, _, err := h.workspace.readSnapshot()
	if err != nil {
		h.writeError(w, r, http.StatusInternalServerError, err.Error())
		return
	}
	var node *workspaceNode
	var clusterID string
outer:
	for pi := range projects {
		for ci := range projects[pi].Clusters {
			for ni := range projects[pi].Clusters[ci].Nodes {
				if projects[pi].Clusters[ci].Nodes[ni].ID == nodeID {
					node = &projects[pi].Clusters[ci].Nodes[ni]
					clusterID = projects[pi].Clusters[ci].ID
					break outer
				}
			}
		}
	}
	if node == nil {
		h.writeError(w, r, http.StatusNotFound, "node not found")
		return
	}

	// Register config with connection manager
	h.connMgr.Register(nodeID, connection.Config{
		Host:     node.Host,
		Port:     node.Port,
		User:     node.User,
		Password: node.Password,
		Database: node.Database,
	})

	// Update status to connecting
	_ = h.workspace.UpdateNodeStatus(nodeID, "connecting", "")

	// Activate
	if err := h.connMgr.Activate(nodeID); err != nil {
		_ = h.workspace.UpdateNodeStatus(nodeID, "failed", err.Error())
		h.writeError(w, r, http.StatusBadGateway, "failed to connect: "+err.Error())
		return
	}

	_ = h.workspace.UpdateNodeStatus(nodeID, "ready", "")
	client, _ := h.connMgr.Get(nodeID)
	version, _ := client.GetVersion()
	dataDir, _ := client.GetPGDataDir()

	writeJSON(w, r, http.StatusOK, map[string]any{
		"success":   true,
		"nodeId":    nodeID,
		"status":    "ready",
		"version":   version,
		"dataDir":   dataDir,
	})
}

// ServeNodeStatus handles GET /api/nodes/{id}/status.
func (h *Handler) ServeNodeStatus(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodGet {
		h.writeError(w, r, http.StatusMethodNotAllowed, "GET required")
		return
	}
	nodeID := extractNodeID(r.URL.Path)
	if nodeID == "" {
		h.writeError(w, r, http.StatusBadRequest, "node id required")
		return
	}

	// Get status from workspace
	projects, _, err := h.workspace.readSnapshot()
	if err != nil {
		h.writeError(w, r, http.StatusInternalServerError, err.Error())
		return
	}
	var node *workspaceNode
	for pi := range projects {
		for ci := range projects[pi].Clusters {
			for ni := range projects[pi].Clusters[ci].Nodes {
				if projects[pi].Clusters[ci].Nodes[ni].ID == nodeID {
					node = &projects[pi].Clusters[ci].Nodes[ni]
					break
				}
			}
		}
	}
	if node == nil {
		h.writeError(w, r, http.StatusNotFound, "node not found")
		return
	}

	connected, err := h.connMgr.Health(nodeID)
	resp := map[string]any{
		"nodeId":            nodeID,
		"connectionStatus":  node.ConnectionStatus,
		"lastError":        node.LastError,
		"connected":         connected,
	}
	if err != nil {
		resp["error"] = err.Error()
	}

	writeJSON(w, r, http.StatusOK, resp)
}

// ServeNodeDeactivate handles POST /api/nodes/{id}/deactivate.
func (h *Handler) ServeNodeDeactivate(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		h.writeError(w, r, http.StatusMethodNotAllowed, "POST required")
		return
	}
	nodeID := extractNodeID(r.URL.Path)
	if nodeID == "" {
		h.writeError(w, r, http.StatusBadRequest, "node id required")
		return
	}

	_ = h.connMgr.Deactivate(nodeID)
	_ = h.workspace.UpdateNodeStatus(nodeID, "unknown", "")

	writeJSON(w, r, http.StatusOK, map[string]any{
		"success": true,
		"nodeId":  nodeID,
		"status":  "unknown",
	})
}

// ServeNodeRegister handles POST /api/nodes/{id}/register.
// Registers a node's connection config without activating.
func (h *Handler) ServeNodeRegister(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		h.writeError(w, r, http.StatusMethodNotAllowed, "POST required")
		return
	}
	nodeID := extractNodeID(r.URL.Path)
	if nodeID == "" {
		h.writeError(w, r, http.StatusBadRequest, "node id required")
		return
	}

	body, err := io.ReadAll(http.MaxBytesReader(w, r.Body, 64*1024))
	if err != nil {
		h.writeError(w, r, http.StatusBadRequest, "failed to read body")
		return
	}
	defer r.Body.Close()

	var req struct {
		Host     string `json:"host"`
		Port     int    `json:"port"`
		User     string `json:"user"`
		Password string `json:"password"`
		Database string `json:"database"`
	}
	if err := json.Unmarshal(body, &req); err != nil {
		h.writeError(w, r, http.StatusBadRequest, "invalid JSON")
		return
	}

	h.connMgr.Register(nodeID, connection.Config{
		Host:     req.Host,
		Port:     req.Port,
		User:     req.User,
		Password: req.Password,
		Database: req.Database,
	})

	writeJSON(w, r, http.StatusOK, map[string]any{"success": true})
}

func extractNodeID(path string) string {
	parts := strings.TrimPrefix(path, "/api/nodes/")
	i := strings.Index(parts, "/")
	if i >= 0 {
		return parts[:i]
	}
	return strings.TrimSpace(parts)
}

const activateNode = async (nodeId: string) => {
  const res = await fetch(`/api/nodes/${nodeId}/activate`, { method: 'POST' })
  const data = await res.json()
  if (data.success) {
    usePGStore.getState().setActiveNodeId(nodeId)
    usePGStore.getState().setConnected(true)
    usePGStore.getState().setVersion(data.version || '')
    usePGStore.getState().setDataDir(data.dataDir || '')
  }
}
