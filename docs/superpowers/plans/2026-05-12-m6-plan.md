```go
// StartPhysicalReplica 启动物理复制集群
func (s *Service) StartPhysicalReplica(ctx context.Context, spec ReplicationSpec) (ReplicaInfo, error) {
    if spec.PrimaryPort == 0 {
        spec.PrimaryPort = 5432
    }
    if spec.SecondaryPort == 0 {
        spec.SecondaryPort = 5433
    }

    p, ok := s.replicationProviders[spec.ProviderID]
    if !ok {
        return ReplicaInfo{}, &ErrProviderUnavailable{Provider: spec.ProviderID, Reason: "not registered"}
    }

    // 首先启动 primary
    primarySpec := InstanceSpec{
        Name:      spec.Name + "-primary",
        PGVersion: spec.PGVersion,
        Port:      spec.PrimaryPort,
        DataDir:   "/var/lib/postgresql/data",
    }
    primaryInfo, err := p.Start(ctx, primarySpec)
    if err != nil {
        return ReplicaInfo{}, fmt.Errorf("failed to start primary: %w", err)
    }

    // 等待 primary 就绪
    if err := waitForPostgres(ctx, primaryInfo.Host, primaryInfo.Port); err != nil {
        return ReplicaInfo{}, fmt.Errorf("primary not ready: %w", err)
    }

    // 启动 standby
    replicaInfo, err := p.StartReplica(ctx, spec, primaryInfo)
    if err != nil {
        // 清理 primary
        p.Stop(ctx, primaryInfo)
        return ReplicaInfo{}, fmt.Errorf("failed to start replica: %w", err)
    }

    return replicaInfo, nil
}

// StopReplicaCluster 停止整个复制集群
func (s *Service) StopReplicaCluster(ctx context.Context, replicaInfo ReplicaInfo) error {
    p, ok := s.replicationProviders[replicaInfo.ProviderID]
    if !ok {
        return &ErrProviderUnavailable{Provider: replicaInfo.ProviderID, Reason: "not registered"}
    }
    return p.StopReplica(ctx, replicaInfo)
}
```

```go
func waitForPostgres(ctx context.Context, host string, port int) error {
    addr := fmt.Sprintf("%s:%d", host, port)
    for i := 0; i < 30; i++ {
        select {
        case <-ctx.Done():
            return ctx.Err()
        default:
        }
        conn, err := net.DialTimeout("tcp", addr, time.Second)
        if err == nil {
            conn.Close()
            return nil
        }
        time.Sleep(time.Second)
    }
    return fmt.Errorf("postgres not ready at %s", addr)
}
```

```go
func (g *ComposeGenerator) GeneratePhysicalCompose(projectName, pgVersion, password string, primaryPort, secondaryPort int) (string, error) {
    tmpl := `version: '3.8'
services:
  primary:
    image: postgres:{{.PGVersion}}
    environment:
      POSTGRES_PASSWORD: {{.Password}}
      POSTGRES_USER: postgres
    ports:
      - "{{.PrimaryPort}}:5432"
    volumes:
      - primary_data:/var/lib/postgresql/data
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 5s
      timeout: 5s
      retries: 5
    command: >
      postgres
      -c wal_level=replica
      -c max_wal_senders=10
      -c hot_standby=on
  standby:
    image: postgres:{{.PGVersion}}
    environment:
      POSTGRES_PASSWORD: {{.Password}}
      POSTGRES_USER: postgres
    ports:
      - "{{.SecondaryPort}}:5432"
    volumes:
      - standby_data:/var/lib/postgresql/data
    depends_on:
      primary:
        condition: service_healthy
    command: >
      postgres
      -c primary_conninfo=host=primary port=5432 user=postgres password={{.Password}}
      -c hot_standby=on
volumes:
  primary_data:
  standby_data:
`
    data := struct {
        PGVersion    string
        Password     string
        PrimaryPort int
        SecondaryPort int
    }{
        PGVersion:    pgVersion,
        Password:     password,
        PrimaryPort:  primaryPort,
        SecondaryPort: secondaryPort,
    }

    var buf bytes.Buffer
    tmplParsed, err := template.New("physical").Parse(tmpl)
    if err != nil {
        return "", fmt.Errorf("failed to parse template: %w", err)
    }
    if err := tmplParsed.Execute(&buf, data); err != nil {
        return "", fmt.Errorf("failed to execute template: %w", err)
    }

    return buf.String(), nil
}
```

```go
import (
    "context"
    "os"
    "os/exec"
    "path/filepath"

// ComposeRunner 执行 docker compose 命令
type ComposeRunner struct {
    workDir string
}

// NewComposeRunner 创建新的 Runner
func NewComposeRunner(workDir string) *ComposeRunner {
    return &ComposeRunner{workDir: workDir}
}

// Up 创建并启动容器
func (r *ComposeRunner) Up(ctx context.Context, projectName, composeFile string) error {
    composePath := filepath.Join(r.workDir, projectName+".yml")
    if err := os.WriteFile(composePath, []byte(composeFile), 0644); err != nil {
        return fmt.Errorf("failed to write compose file: %w", err)
    }

    cmd := exec.CommandContext(ctx, "docker", "compose",
        "-f", composePath,
        "-p", projectName,
        "up", "-d")
    cmd.Dir = r.workDir
    output, err := cmd.CombinedOutput()
    if err != nil {
        return fmt.Errorf("docker compose up failed: %w, output: %s", err, output)
    }
    return nil
}

// Down 停止并销毁容器
func (r *ComposeRunner) Down(ctx context.Context, projectName, composeFile string) error {
    composePath := filepath.Join(r.workDir, projectName+".yml")
    cmd := exec.CommandContext(ctx, "docker", "compose",
        "-f", composePath,
        "-p", projectName,
        "down", "-v")
    cmd.Dir = r.workDir
    output, err := cmd.CombinedOutput()
    if err != nil {
        return fmt.Errorf("docker compose down failed: %w, output: %s", err, output)
    }
    return nil
}
```

```go
package provision

import (
    "context"
    "fmt"
    "time"
)

// DockerReplicationProvider 使用 Docker 实现复制
type DockerReplicationProvider struct {
    composeGenerator *ComposeGenerator
    composeRunner    *ComposeRunner
    workDir           string
}

// NewDockerReplicationProvider 创建新的 DockerReplicationProvider
func NewDockerReplicationProvider(workDir string) *DockerReplicationProvider {
    return &DockerReplicationProvider{
        composeGenerator: NewComposeGenerator(),
        composeRunner:    NewComposeRunner(workDir),
        workDir:          workDir,
    }
}

// ID 返回 provider ID
func (p *DockerReplicationProvider) ID() string {
    return "docker-replication"
}

// StartReplica 启动主备或逻辑复制集群
func (p *DockerReplicationProvider) StartReplica(ctx context.Context, spec ReplicationSpec, primaryInfo InstanceInfo) (ReplicaInfo, error) {
    projectName := fmt.Sprintf("pgv-replica-%s-%d", spec.Name, time.Now().UnixMilli())
    password := "postgres" // 默认密码

    var composeFile string
    var err error

    if spec.Type == "physical" {
        composeFile, err = p.composeGenerator.GeneratePhysicalCompose(
            projectName, spec.PGVersion, password,
            spec.PrimaryPort, spec.SecondaryPort,
        )
    } else {
        composeFile, err = p.composeGenerator.GenerateLogicalCompose(
            projectName, spec.PGVersion, password,
            spec.PrimaryPort, spec.SecondaryPort,
        )
    }
    if err != nil {
        return ReplicaInfo{}, fmt.Errorf("failed to generate compose: %w", err)
    }

    // 启动容器
    if err := p.composeRunner.Up(ctx, projectName, composeFile); err != nil {
        return ReplicaInfo{}, fmt.Errorf("failed to start containers: %w", err)
    }

    // 等待 secondary 就绪
    secondaryHost := "localhost"
    if err := waitForPostgres(ctx, secondaryHost, spec.SecondaryPort); err != nil {
        p.composeRunner.Down(ctx, projectName, composeFile)
        return ReplicaInfo{}, fmt.Errorf("secondary not ready: %w", err)
    }

    return ReplicaInfo{
        ProviderID:    p.ID(),
        ComposeProject: projectName,
        PrimaryInfo:   primaryInfo,
        SecondaryInfo: InstanceInfo{
            ProviderID: "docker",
            Host:       secondaryHost,
            Port:       spec.SecondaryPort,
            Name:       spec.Name + "-secondary",
        },
    }, nil
}

// StopReplica 停止复制集群
func (p *DockerReplicationProvider) StopReplica(ctx context.Context, replicaInfo ReplicaInfo) error {
    // 获取原始 compose file content 需要保存，这里简化处理
    // 实际实现需要在 StartReplica 时保存 compose file 到磁盘
    composeFile := "" // 从 replicaInfo 中恢复或重新生成
    return p.composeRunner.Down(ctx, replicaInfo.ComposeProject, composeFile)
}

// GetReplicationStatus 获取复制状态
func (p *DockerReplicationProvider) GetReplicationStatus(ctx context.Context, replicaInfo ReplicaInfo) (ReplicationStatus, error) {
    // 查询 primary 的复制状态
    // 使用 docker exec 或直接连接 postgres 查询
    // 简化实现：检查 secondary 是否可达
    secondaryAddr := fmt.Sprintf("%s:%d", replicaInfo.SecondaryInfo.Host, replicaInfo.SecondaryInfo.Port)

    conn, err := net.DialTimeout("tcp", secondaryAddr, 5*time.Second)
    if err != nil {
        return ReplicationStatus{
            PrimaryConnected:  true,
            SecondaryConnected: false,
            ReplicationWorking: false,
            LAG: "unknown",
        }, nil
    }
    conn.Close()

    return ReplicationStatus{
        PrimaryConnected:   true,
        SecondaryConnected: true,
        ReplicationWorking: true,
        LAG:                "0 B",
        LastHeartbeat:      time.Now().UnixMilli(),
    }, nil
}
```

```go
func (h *Handler) serveProvisionReplica(w http.ResponseWriter, r *http.Request) {
    ctx := r.Context()

    var req ReplicaProvisionRequest
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
        writeError(w, http.StatusBadRequest, "invalid request: %s", err)
        return
    }

    // 验证请求
    if req.ProjectID == "" || req.ClusterName == "" || req.Type == "" {
        writeError(w, http.StatusBadRequest, "missing required fields")
        return
    }
    if req.Type != "physical" && req.Type != "logical" {
        writeError(w, http.StatusBadRequest, "type must be 'physical' or 'logical'")
        return
    }

    project, err := h.wsStore.GetProject(ctx, req.ProjectID)

    cluster := &WorkspaceCluster{
```
